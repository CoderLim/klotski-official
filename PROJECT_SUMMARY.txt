    ├── engine/          # 游戏引擎
    │   ├── collision   # 碰撞检测

    │   ├── validation  # 边界验证
    │   └── win         # 胜利判定
    ├── puzzles/         # 拼图数据

    ├── store/           # 状态管理

    └── utils/           # 工具函数
